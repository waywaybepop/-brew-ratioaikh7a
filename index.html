document.addEventListener("DOMContentLoaded", () => { const background = document.body; let x = 0, y = 0; let speedX = (Math.random() - 0.5) * 1.5; let speedY = (Math.random() - 0.5) * 1.5; function animateBackground() { x += speedX; y += speedY; if (Math.abs(x) > 10) speedX *= -1; if (Math.abs(y) > 10) speedY *= -1; background.style.backgroundPosition = `${x}px ${y}px`; requestAnimationFrame(animateBackground); } animateBackground(); });
